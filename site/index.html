🧩 Use as a Library

import { TradingEngine, ClawVault, TamaGOchi } from "nanosolana";

// OODA trading engine
const engine = new TradingEngine(config);
engine.on("signal", signal => console.log(signal));

// 3-tier memory
const vault = new ClawVault();
vault.storeKnown({ content: "SOL at $142.50", source: "birdeye", tags: ["SOL"] });

// Virtual pet
const pet = new TamaGOchi("MyPet");
console.log(pet.getStatusDisplay());
